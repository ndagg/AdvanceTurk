        m_table.style.display = "block";
        p_table.style.display = "none";
        v_table.style.display = "none";

          grantVisionMenu.style.display = 'none';

      }

    function ToggleInfo() {
      if (p_table.style.display === "none") {
        p_table.style.display = "block";
        v_table.style.display = "none";

        if(grantVisionMenu && grantVisionMenuToggle) {
          grantVisionMenu.style.display = 'none';

      else {
        p_table.style.display = "none";
      }

      if (v_table.style.display === "none") {
        v_table.style.display = "block";
        p_table.style.display = "none";

        if(grantVisionMenu && grantVisionMenuToggle) {

      else {
          v_table.style.display = "none";
      }

      const grantVisionSelect = document.getElementById("grant-vision-select");

      const removeVisionBtn = document.getElementById("remove-vision-btn");
      const removeFriendSelector = document.getElementById("remove-vision-selector");
      const confirmGrantMenu = document.getElementById("grant-vision-confirmation");
      const confirmGrant = document.getElementById("grant-confirm");
      const confirmScreenFriend = document.getElementById("grant-vision-conf-friend");
      const grantCancel = document.getElementById("grant-cancel");

      grantVisionMenuToggle.onclick = function() {
        const menuDisplay = grantVisionMenu.style.display;
        applyCSS(grantVisionMenu, {
          display: (function() {
            if(menuDisplay === "none") {
              return "flex";
            } else {
              return "none";

        updateVisionList("add", addFriendSelector, removeFriendSelector);

            updateVisionList("remove", removeFriendSelector, addFriendSelector);

        const  friend = selector.value;
        const userInfo = {
          action: action,
          gamesId: 1432305,
          selectedFriend: friend
        }

          grantVisionMsg.textContent = err;
        })
